59void transmitterMain()
60{
66 switch(transmit_web){
67 case THINGSPEAK_AIRQUALITY:
68 tempHumi_read(&temperature, &humidity);
69 airQual_measurement(&temperature, &humidity);
70 dataLCD_print(airQual_get_Value(), THINGSPEAK_AIRQUALITY);
71 sendData_fieldValue(THINGSPEAK_AIRQUALITY, airQual_get_Value());
72 break;
73
74 case THINGSPEAK_MOTION:
75 sendData_fieldValue(THINGSPEAK_MOTION, motion_val);
76 break;
77
78 case THINGSPEAK_TEMPERATURE:
79 tempHumi_read(&temperature, &humidity);
80 dataLCD_print(temperature, THINGSPEAK_TEMPERATURE);
81 sendData_fieldValue(THINGSPEAK_TEMPERATURE, temperature);
82 break;
83
84 case THINGSPEAK_HUMIDITY:
85 tempHumi_read(&temperature, &humidity);
86 dataLCD_print(humidity, THINGSPEAK_HUMIDITY);
87 sendData_fieldValue(THINGSPEAK_HUMIDITY, humidity);
88 break;
89
90 default:
91 break;
92 }
93
94 // Resets counter, so each sensor/module updates
95 if(transmit_web >= THINGSPEAK_TRANSMITMAX){
96 transmit_web = 0;
97 }
98
99 // Extra motion sensor update, to ensure motion detection is detected
100 motionSensor_detect(&motion_state, &motion_val);
101 delay(seconds_30); // Thingspeak delay (15 seconds), 30 seconds just in case
102 transmit_web++;
103}
